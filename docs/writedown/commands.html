 22        """
 23        Returns a generator that, for each structural node in the current document, yields a tuple that contains:
 24        1. The level of the node
 25        2. The node itself
 26        3. A dictionary of all the characters occurring within the scope of the node which were defined by a Character instruction
 27           in the document.  The dictionary key is the CharacterNode and the value is the number of times the Character
 28           was referenced.
 29
 30        Example:
 31        (3, ChapterNode, { CharacterNode: 2, CharacterNode: 3 })
 32        """
 33        character_defs = [character for character in self.doc.find(CharacterNode)]
 34        yield from self._characters(character_defs, self.doc, 0)

 36    def _characters(self, character_defs:Iterable[CharacterNode], node_holder:ASTNode, level:int) -> Generator[Tuple[int, ASTNode, Dict[CharacterNode, int]], None, None]:
 37        """
 38        Returns a generator that, for each structural node in node_holder, yields a tuple that contains:
 39        1. The level of the node
 40        2. The node itself
 41        3. A dictionary of all the characters occurring within the scope of the node which are defined in character_defs.
 42           The dictionary key is the CharacterNode and the value is the number of times the Character was referenced.
 43
 44        Example:
 45        (3, ChapterNode, { CharacterNode: 2, CharacterNode: 3 })
 46        """
 47        def tally_push(characters:dict, character_def:CharacterNode, count:int):
 48            if character_def not in characters:
 49                characters[character_def] = 0
 50            characters[character_def] += count
 51
 52        if node_holder.is_structural_node():
 53            characters = {}
 54            for text in node_holder.find(TextNode, recursive=node_holder.is_structural_leaf_node()):
 55                for character_def in character_defs:
 56                    if character_def.name in text.text:
 57                        tally_push(characters, character_def, text.text.count(character_def.name))
 58                    for name_form in character_def.name_forms:
 59                        if name_form in text.text:
 60                            tally_push(characters, character_def, text.text.count(name_form))
 61            yield ((level, node_holder, characters))
 62
 63        for node in node_holder.nodes:
 64            yield from self._characters(character_defs, node, level + 1)

169    def _locations(self, place_defs:Iterable[PlaceNode], node_holder:ASTNode, level:int) -> Generator[Tuple[int, ASTNode, Dict[LocationNode, PlaceNode]], None, None]:
170        """
171        Returns a generator that, for each structural node in node_holder, yields a tuple that contains:
172        1. The level of the node
173        2. The node itself
174        3. A dictionary of the locations occurring within the scope of the node.  The dictionary key is the 
175           LocationNode and the value is the resolved PlaceNode of the location if one exists in place_defs.
176
177        Example:
178        (3, ChapterNode, { CharacterNode: 2, CharacterNode: 3 })
179        """
180        if node_holder.is_structural_node():
181            locations = {}
182            for location in node_holder.find(LocationNode, recursive=node_holder.is_structural_leaf_node()):
183                if location not in locations:
184                    place = None
185                    for place_def in place_defs:
186                        #if place_def.path() == location.path():
187                        if place_def.name == location.name:
188                            place = place_def
189                            break
190                    locations[location] = place
191            yield ((level, node_holder, locations))
192
193        for node in node_holder.nodes:
194            yield from self._locations(place_defs, node, level + 1)

210    def _sessions(self, node_holder:ASTNode, map:dict, last:List[ASTNode], level:int) -> Generator[dict, None, None]:
211        """
212        Returns a generator that yields an aggregated dictionary of the following structure for each SessionNode encountered in 
213        node_holder's hierarchy that differs from the session stored in the first element of last:
214
215        {
216            'session': session string representation,
217            'entry': the first node the session applies to,
218            'target': the session target wordcount,
219            'wordcount': the actual wordcount of the session scope,
220            'delta': the difference between the target and actual wordcounts
221        }
222
223        The first element of last is manipulated by reference to maintain and track the last session encountered.
224        """
225        for node in node_holder.nodes:
226            if node.session:
227                session = node.session.__str__()
228
229                if not last[0]:
230                    last[0] = session
231
232                if not session in map:
233                    map[session] = { 
234                        'session': session, 
235                        'entry': node, 
236                        'target': node.session.target, 
237                        'wordcount': 0,
238                        'delta': 0 
239                    }
240
241                map[session]['wordcount'] += node.wordcount(False)
242                map[session]['delta'] = map[session]['wordcount'] - map[session]['target'] if map[session]['target'] else None
243
244                if last[0] != session:
245                    yield map[last[0]]
246                    del map[last[0]]
247                    last[0] = session
248            else:
249                if last[0] in map:
250                    yield map[last[0]]
251                    del map[last[0]]
252                    last[0] = None
253
254            yield from self._sessions(node, map, last, level + 1)
255
256        if level == 0 and last[0] in map:
257            yield map[last[0]]
258            del map[last[0]]

341    def _targets(self, node_holder:ASTNode, level:int) -> Generator[Tuple[int, ASTNode, int, int, int], None, None]:
342        """
343        Returns a generator that, for each structural node in node_holder, yields a tuple that contains:
344        1. The level of the node
345        2. The node itself
346        3. The target wordcount
347        4. The actual wordcount
348        5. The delta wordcount (difference between target and actual)
349
350        Example:
351        (3, ChapterNode, 1000, 750, 250)
352        """
353        if node_holder.is_structural_node():
354            target = sum(node.value for node in node_holder.find(TargetNode, recursive=node_holder.is_structural_leaf_node()))

356            wordcount = node_holder.wordcount()
357            delta = wordcount - target if target else None
358            yield (level, node_holder, target, wordcount, delta)
359
360        for node in node_holder.nodes:
361            yield from self._targets(node, level + 1)

363    def todo(self, context:bool=False) -> Generator[Tuple[int, ASTNode, bool], None, None]:
364        """
365        Returns a generator that, for each TodoNode in the current document, yields a tuple that contains:
366        1. The level of the node
367        2. The node itself
368        3. A boolean value indicating if this node is a contextual node or not
369
370        If context is False, only TodoNodes will be yielded.
371        If context is True, the node immediately preceding and the node immediately proceeding each TodoNode
372        will be yielded in addition to each TodoNode.
373
374        Examples:
375        (4, TodoNode, False)
376
377        With context set to True:
378        (4, TextNode, True)
379        (4, TodoNode, False)
380        (4, TextNode, True)
381        """
382        emitted = []
383        nodes = self.doc.find(TodoNode)
384        for node in nodes:
385            level = 0
386            if not node.parent in emitted:
387                parents = []
388                parent = node.parent
389                while parent != None:
390                    parents.append(parent)
391                    parent = parent.parent
392
393                parents.reverse()
394                for parent in parents:
395                    if (not parent in emitted and parent.is_structural_node()):
396                        yield (level, parent, False)
397                        emitted.append(parent)
398                    level += 1
399
400            if context:
401                index = node.parent.nodes.index(node)
402                if index - 1 >= 0:
403                    sibling = node.parent.nodes[index - 1]
404                    yield (level, sibling, True)
405                elif node.parent:
406                    yield (level, parent, True)
407
408            yield (level, node, False)
409
410            if context:
411                index = node.parent.nodes.index(node)
412                if index + 1 < len(node.parent.nodes):
413                    sibling = node.parent.nodes[index + 1]
414                    yield (level, sibling, True)

212        """
213        Returns a generator that yields an aggregated dictionary of the following structure for each SessionNode encountered in 
214        node_holder's hierarchy that differs from the session stored in the first element of last:

217            'session': session string representation,

245                if last[0] != session:
246                    yield map[last[0]]

248                    last[0] = session
249            else:
250                if last[0] in map:
251                    yield map[last[0]]
252                    del map[last[0]]
253                    last[0] = None
