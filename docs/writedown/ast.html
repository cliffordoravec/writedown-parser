 54class ASTNode(ASTComponent):
 55    """Represents an abstract syntax tree node."""
 56
 57    def __init__(self, sourceline:SourceLine):
 58        """Initializes a new ASTNode instance based on the given sourceline."""
 59
 60        self.parent:ASTNode = None
 61        """This node's parent."""
 62
 63        self.nodes:List[ASTNode] = []
 64        """Children of this node."""
 65
 66        self.session:SessionAttribute = None
 67        """The session this node is associated with."""
 68
 69        self.session_template:SessionAttribute = None
 70        """
 71        Used to assign a child node's session when a child node is appended to this node.
 72
 73        This is used during parsing to avoid assigning to or overwriting this node's session
 74        when this node's children belong to a session that this node does not.
 75        """
 76
 77        super().__init__(sourceline)
 78
 79    def append(self, child:'ASTNode'):
 80        """
 81        Appends child to this node's children.
 82
 83        If this node has a session template defined, it will be assigned as the child node's
 84        session.  Otherwise, if this node has a session defined, it will be assigned as the
 85        child node's session.
 86        """
 87        child.parent = self
 88        if self.session_template:
 89            child.session = self.session_template
 90        elif self.session:
 91            child.session = self.session
 92        self.nodes.append(child)
 93    
 94    def extend(self, children:Iterator['ASTNode']):
 95        """Appends each of the nodes in children to this node's children."""
 96        for child in children:
 97            self.append(child)
 98
 99    def filter(self, func:Callable, recursive:bool=True) -> Generator['ASTNode', None, None]:
100        """
101        Returns a generator of this node's descendents that func(child) returns True for. 
102
103        If recursive is True, all descendents will be processed; if False, only this node's
104        immediate children will be processed.
105        """
106        for child in self.nodes:
107            if func(child):
108                yield child
109            if recursive:
110                yield from child.filter(func, recursive=recursive)
111
112    def find(self, node_type:type, recursive:bool=True) -> Generator['ASTNode', None, None]:
113        """
114        Returns a generator of this node's descendents that are an instance of node_type.
115
116        If recursive is True, all descendents will be processed; if False, only this node's
117        immediate children will be processed.
118        """
119        for child in self.nodes:
120            if isinstance(child, node_type):
121                yield child
122            if recursive:
123                yield from child.find(node_type, recursive=recursive)
124
125    def is_structural_node(self) -> bool:
126        """Returns True if the node is an instance of a structural node (DocumentNode, ActNode, PartNode, ChapterNone, SceneNode, or SectionNode) and False if not."""
127        return (isinstance(self, DocumentNode)
128            or isinstance(self, ActNode)
129            or isinstance(self, PartNode)
130            or isinstance(self, ChapterNode)
131            or isinstance(self, SceneNode)
132            or isinstance(self, SectionNode))
133
134    def is_structural_leaf_node(self) -> bool:
135        """Returns True if the node is the last instance of a structural node in its hierarchy and False if not."""
136        if not self.is_structural_node(): return False
137        try:
138            next(self.filter(lambda child: child.is_structural_node))
139        except StopIteration:
140            return True
141        return False 
142
143    def structural_lineage(self) -> List['ASTNode']:
144        """
145        Returns a list of the structural nodes in this node's lineage, including this node.
146        """
147        lineage = []
148        current = self
149        while current != None:
150            if current.is_structural_node():
151                lineage.append(current)
152            current = current.parent
153        lineage.reverse()
154        return lineage
155
156    def structural_path(self) -> str:
157        """
158        Returns a formatted string representing this node's lineage in the form of "Grandparent > Parent > Self".
159        """
160        return ' > '.join(node.__str__() for node in self.structural_lineage())
161
162    def wordcount(self, recursive:bool=True) -> int:
163        """
164        Returns the wordcount of all TextNodes in this node's hierarchy, including this node.
165
166        If recursive is True, all descendents will be processed; if False, only this node
167        will be processed.
168        """
169        if isinstance(self, TextNode): return len(self.text.split())
170        if recursive:
171            return sum(len(child.text.split()) for child in self.find(TextNode, recursive=recursive))
172        return 0
173
174    def charcount(self, recursive:bool=True) -> int:
175        """
176        Returns the character count of all TextNodes in this node's hierarchy, including this node.
177
178        If recursive is True, all descendents will be processed; if False, only this node
179        will be processed.
180        """
181        if isinstance(self, TextNode): return len(self.text)
182        if recursive:
183            return sum(len(child.text) for child in self.find(TextNode, recursive=recursive))
184        return 0

101        """

104        If recursive is True, all descendents will be processed; if False, only this node's
105        immediate children will be processed.
106        """
107        for child in self.nodes:

136        """Returns True if the node is the last instance of a structural node in its hierarchy and False if not."""
137        if not self.is_structural_node(): return False
