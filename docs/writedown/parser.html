 20    def __init__(self, sequences:Dict[str, int]=None):

 24    def sequence_key(self, node_holder:ASTNode, structural_type:str) -> str:
 25        """
 26        Returns a dict key for tracking sequences for the given structural type in the context of the given node holder.
 27
 28        The structural_type is a string because this method should be invoked prior to creation of the corresponding ASTNode instance.
 29        """
 30        return f'{node_holder.structural_path()} [{structural_type}]'

 32    def next_sequence(self, node_holder:ASTNode, structural_type:str) -> int:
 33        """
 34        Returns the next sequence number for the given structural type in the context of the given node holder.
 35
 36        The structural_type is a string because this method should be invoked prior to creation of the corresponding ASTNode instance.
 37        """
 38        key = self.sequence_key(node_holder, structural_type)
 39        if not key in self.sequences:
 40            self.sequences[key] = 0
 41        self.sequences[key] += 1
 42        return self.sequences[key]

 44    def set_sequence(self, node_holder:ASTNode, structural_type:str, number:int) -> None:
 45        """
 46        Sets the current sequence number for the given structural type in the context of the given node holder.
 47
 48        The structural_type is a string because this method should be invoked prior to creation of the corresponding ASTNode instance.
 49        """
 50        key = self.sequence_key(node_holder, structural_type)
 51        if key in self.sequences:
 52            existing = self.sequences[key]
 53            # TODO: Pass the source line and include that information in the warning message:
 54            if existing > number:
 55                warnings.warn(f'{structural_type} {number}: Sequence is less than previous sequence {structural_type} {existing}.', SequenceWarning)
 56            elif existing == number:
 57                warnings.warn(f'{structural_type} {number}: Sequence is the same as previous sequence {structural_type} {existing}.', SequenceWarning)
 58            elif existing < number - 1:
 59                warnings.warn(f'{structural_type} {number}: Sequence contains a gap from previous sequence {structural_type} {existing}.', SequenceWarning)
 60        self.sequences[key] = number

 62    def get_or_set_sequence(self, node_holder:ASTNode, structural_type:str, number:int) -> int:
 63        """
 64        Gets the next or sets the current sequence number for the given structrual type in the context of the given node holder
 65        based on the value of number.
 66
 67        If number is None, the next sequence number is retrieved.  Otherwise, number is set as the current sequence number.
 68        """
 69        if number == None:
 70            return self.next_sequence(node_holder, structural_type)
 71        self.set_sequence(node_holder, structural_type, number)
 72        return number

 74    def strip_instruction(self, str:str) -> str:
 75        """
 76        Removes leading Writedown instructions from a string.
 77
 78        For example, "@chapter One" would be returned as "One".
 79        """
 80        regex = '^\\' + Tokens.INSTRUCTION + '[^\s]+\s*'

 87    def get_instruction(self, str:str) -> str:
 88        """
 89        Returns the leading Writedown instruction token for the given string; returns None if one is not present.
 90
 91        For example, "@chapter One" would return "@chapter".
 92        """

106    def parse(self, node_holder:ASTNode, lines:IndexedGeneratorBuffer, pos:int=0, end:int=None) -> int:
107        """
108        Parses the supplied lines for Writedown markup, starting and ending at the provided index positions
109        (inclusive and exclusive, respectively), in the context of the given node holder.  
110
111        Returns the (unprocessed) index from which subsequent parsing should resume, if any.
112        """
113        while lines.valid_index(pos) and (end == None or end > pos):
114            line = lines.get(pos)
115
116            if self.is_instruction(line.content, Tokens.INSTRUCTION):
117                instruction = self.get_instruction(line.content)
118
119                match instruction:
120                    case Tokens.EOF:
121                        self.set_session_template(node_holder, None)
122                        pos += 1
123
124                    case Tokens.ACT:
125                        pos = ActParser(self.sequences).parse(node_holder, lines, pos)
126
127                    case Tokens.AUTHOR:
128                        node_holder.append(AuthorNode(line, self.strip_instruction(line.content)))
129                        pos += 1
130
131                    case Tokens.CHAPTER:
132                        pos = ChapterParser(self.sequences).parse(node_holder, lines, pos)
133
134                    case Tokens.CHARACTER:
135                        pos = CharacterParser(self.sequences).parse(node_holder, lines, pos)
136
137                    case Tokens.COMMENT | Tokens.COMMENT_SHORTHAND:
138                        node_holder.append(CommentNode(line, self.strip_instruction(line.content)))
139                        pos += 1
140
141                    case Tokens.COMMENT_BLOCK_START:
142                        pos = CommentBlockParser(self.sequences).parse(node_holder, lines, pos)
143
144                    case Tokens.END_SESSION:
145                        self.set_session_template(node_holder, None)
146                        pos += 1
147
148                    case Tokens.INCLUDE:
149                        path = self.strip_instruction(line.content)
150                        if line.source != 'string':
151                            path = os.path.join(os.path.dirname(line.source), path)
152                        # Must create a new Parser instance here in case an @include is being processed
153                        # by one of the subparsers (their parse() method will override this method,
154                        # which is not what we would want):
155                        node_holder.extend(Parser(self.sequences).parse_path(path))
156                        pos += 1
157
158                    case Tokens.LOCATION:
159                        pos = LocationParser(self.sequences).parse(node_holder, lines, pos)
160
161                    case Tokens.PAGEBREAK:
162                        node_holder.append(PageBreakNode(line))
163                        pos += 1
164
165                    case Tokens.PART:
166                        pos = PartParser(self.sequences).parse(node_holder, lines, pos)
167
168                    case Tokens.PLACE:
169                        pos = PlaceParser(self.sequences).parse(node_holder, lines, pos)
170
171                    case Tokens.SCENE:
172                        pos = SceneParser(self.sequences).parse(node_holder, lines, pos)
173
174                    case Tokens.SECTION:
175                        pos = SectionParser(self.sequences).parse(node_holder, lines, pos)
176
177                    case Tokens.SESSION:
178                        session_node = SessionParser(self.sequences).get_session_node(line)
179                        self.set_session_template(node_holder, session_node)
180                        pos += 1
181
182                    case Tokens.STATUS:
183                        node_holder.append(StatusNode(line, self.strip_instruction(line.content)))
184                        pos += 1
185
186                    case Tokens.TABLE_OF_CONTENTS | Tokens.TOC:
187                        node_holder.append(TableOfContentsNode(line))
188                        pos += 1
189
190                    case Tokens.TAG:
191                        tags = self.strip_instruction(line.content).split(' ')
192                        node_holder.append(TagNode(line, tags))
193                        pos += 1
194
195                    case Tokens.TARGET:
196                        wordcount = int(self.strip_instruction(line.content))
197                        node_holder.append(TargetNode(line, wordcount))
198                        pos += 1
199
200                    case Tokens.TITLE:
201                        node_holder.append(TitleNode(line, self.strip_instruction(line.content)))
202                        pos += 1
203
204                    case Tokens.TODO:
205                        node_holder.append(TodoNode(line, self.strip_instruction(line.content)))
206                        pos += 1
207
208                    case _:
209                        node_holder.append(UnmappedInstructionNode(line, instruction, self.strip_instruction(line.content)))
210                        pos += 1
211            else:
212                node_holder.append(TextNode(line, line.content))
213                pos += 1
214
215        # Purge processed lines:
216        lines.truncate(pos - 1)
217
218        return pos

312    def list_path(self, glob_path:str=None, recursive:bool=True) -> Generator[str, None, None]:
313        """
314        Returns a generator that yields the names of files matching the given glob pattern path.
315
316        If a glob pattern path is not provided or if the glob pattern path resolves to a directory, 
317        the following glob patterns will be attempted, and the first successful pattern will be used:
318
319            index.wd
320            **/*.wd
321
322        Raises a ValueError if a non-directory path is specified and no files are matched.
323        """
324        entries = []
325        if glob_path == None or os.path.isdir(glob_path):
326            paths_to_try = ['index.wd', '**/*.wd']
327            for path_to_try in paths_to_try:
328                if glob_path == None:
329                    entries = glob.glob(path_to_try, recursive=recursive)
330                elif os.path.isdir(glob_path):
331                    entries = glob.glob(os.path.join(glob_path, path_to_try), recursive=recursive)
332
333                if len(entries) > 0: break
334        else:
335            entries = glob.glob(glob_path, recursive=recursive)
336
337        if (len(entries) == 0) and not os.path.isdir(glob_path):
338            raise ValueError(f'No files matched path {glob_path}')
339
340        for entry in entries:
341            yield os.fspath(entry)

350        """
351        Parses @act instructions starting at the specified position in the context of the node holder.  The first line must be an @act instruction.
352
353        Returns the (unprocessed) index from which subsequent parsing should resume, if any.
354        """
355
356        instruction_line = lines.get(pos)

364        if match:
365            try:
366                number = match.group(1)
367                if number != None:

370                pass

374            except IndexError:
375                pass

378        act = ActNode(instruction_line, number=number, title=title)
379        node_holder.append(act)

383            line = lines.get(pos)
384            # Read until a new act:
385            if self.is_instruction(line.content, Tokens.ACT):
386                break
387            pos += 1
388
